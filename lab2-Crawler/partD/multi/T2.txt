        function sortByDate(json) {
        	json.sort(function sortJZRQ(a,b){
        		var da = new Date(Date.parse(a.jzsjRq.toString().replace(/-/g,"/"))).getTime();
        		var db = new Date(Date.parse(b.jzsjRq.toString().replace(/-/g,"/"))).getTime();
        		return db - da;
            });
            var minJzRq;
            var newMessageArr = new Array();
        	var oldMessageArr = new Array();
            var newDate = new Date();
        	newDate.setHours(0);
        	newDate.setMinutes(0);
        	newDate.setSeconds(0);
        	newDate.setMilliseconds(0);
            for (var i=0;i<json.length;i++) {
            	var jzsjRq = json[i].jzsjRq;
            	var date = new Date(jzsjRq.toString()).getTime();
            	if (date>=newDate) {
            		newMessageArr.push(json[i]);
            	}else {
            		oldMessageArr.push(json[i]);
            	}
            }
            
            if (newMessageArr.length < 3 && oldMessageArr.length > 0) {
            	for (var j=0;j<oldMessageArr.length;j++) {
            		newMessageArr.push(oldMessageArr[j]);
            		if (newMessageArr.length === 3) {
            			break;
            		}
            	}
            }
        	return newMessageArr;
        }
        
        function findMinJZRQ(json) {
        	var minJzRq = new Date(json[0].jzsjRq.toString()).getTime();
        	for (var i=1;i<json.length;i++) {
        		if(typeof json[i].jzsjRq != 'undefined'){
        			var JzRq = new Date(json[i].jzsjRq.toString()).getTime();
        			if (JzRq < minJzRq) {
        				minJzRq = JzRq;
        			}
        		}
        	}
        	return minJzRq;
        }
        
         function convertMonth(monthStr) {
        	var monthStrAry = ["一月","二月","三月","四月","五月","六月","七月","八月","九月","十月","十一月","十二月"];

        function getTimeArr(json,data) {

        function distinctTimeArr(timeArr) {
           var result = new Array();
           for (var i = 0; i < timeArr.length; i++) {
               var item = timeArr[i];
               if (result.indexOf(item) === -1) {
            	   result.push(item);
               }
           }
           return result;
       }
